⚡ เชื่อมต่อ Google Sheets แบบ Live — ทำครั้งเดียว

  1. เปิด Google Sheets → เมนู Extensions → Apps Script
  2. ลบโค้ดเดิมทั้งหมด แล้ววาง โค้ดนี้ทั้งหมด:
var CSV_URL = "https://docs.google.com/spreadsheets/d/e/2PACX-1vRAe91vfK5wCBTngzgciQST4EiAYTwGWTWsGz1Td-HHAk_z1Nboz-_9JeMztJ9PPglT8AteucZEOcr5/pub?gid=1062233043&single=true&output=csv"; var CACHE_SECS = 300; // cache 5 นาที function doGet(e) { try { // ── ลอง cache ก่อน ── var cache = CacheService.getScriptCache(); var cached = cache.get('csv_data'); if (cached) { return ContentService.createTextOutput(cached) .setMimeType(ContentService.MimeType.TEXT); } // ── ดึงข้อมูลจาก Google Sheets ── var res = UrlFetchApp.fetch(CSV_URL, { muteHttpExceptions: true }); var code = res.getResponseCode(); if (code !== 200) { throw new Error('Google Sheets ตอบกลับ HTTP ' + code + ' — ตรวจสอบว่า Sheet ได้รับการ Publish แล้ว'); } var text = res.getContentText('UTF-8'); if (!text || text.trim().length < 5) { throw new Error('ข้อมูลจาก Sheets ว่างเปล่า — ตรวจสอบ gid และ filter'); } // ── บันทึก cache ── try { cache.put('csv_data', text, CACHE_SECS); } catch (ce) { /* quota เต็ม */ } return ContentService.createTextOutput(text) .setMimeType(ContentService.MimeType.TEXT); } catch (err) { // ── ส่ง JSON error กลับให้ Dashboard แสดงข้อความที่ชัดเจน ── return ContentService.createTextOutput( JSON.stringify({ __gas_error: true, message: err.message }) ).setMimeType(ContentService.MimeType.JSON); } }
  1. กด Deploy → New deployment → Type: Web app
  2. "Who has access" → Anyone → Deploy → คัดลอก URL
  3. วาง URL ด้านล่าง แล้วกดบันทึก
